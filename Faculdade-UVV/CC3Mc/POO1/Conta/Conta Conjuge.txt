package com.mycompany.hansrolf;

public class Conta {
     int numero;
    double saldo;
    double limite = 1000.0;
    Cliente titular;

    void Imprimir(){
        System.out.println("Numero: " + numero);
        System.out.println("Saldo : " + saldo);
        System.out.println("Limite : " +limite);
        System.out.println("Cpf: " + titular.cpf + "\n Nome: "+ titular.nome + "\n sobrenome: " + titular.sobrenome);
    }
    
    void sacar(int saq){
       saldo -= saq;
    }
    void depositar(int dep){
        saldo += dep;
    }

}
